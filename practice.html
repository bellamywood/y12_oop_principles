# Base class (Blueprint) class Card: # Constructor def __init__(self, rank, suit): # Encapsulation: internal attributes self._rank = rank self._suit = suit # Property for controlled access @property def rank(self): return self._rank @property def suit(self): return self._suit # Method shared by all cards def describe(self): return f"{self.rank} of {self.suit}" # Subclass (Inheritance) class FaceCard(Card): # Override method (Polymorphism) def describe(self): return f"Face card: {self.rank} of {self.suit}" # Subclass with custom constructor class Joker(Card): # Use parent constructor def __init__(self): super().__init__("J", "Joker") # Override method def describe(self): return "Joker!" # Create mixed objects deck = [ Card("7", "Hearts"), FaceCard("K", "Spades"), Joker() ] # Polymorphism in action for card in deck: print(card.describe())
